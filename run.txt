    listen 443 ssl http2;
    server_name mckbytes.tech www.mckbytes.tech;

    ssl_certificate /etc/nginx/ssl-certificates/mckbytes.tech.crt;
    ssl_certificate_key /etc/nginx/ssl-certificates/mckbytes.tech.key;

    location /static/ {
        alias /home/mckbytes/htdocs/mckbytes.tech/mckbytes-website/mckbytes-website/media/static/;

    location /media/ {
        alias /home/mckbytes/htdocs/mckbytes.tech/mckbytes-website/media/;
    }

    location / {
        proxy_pass http://127.0.0.1:8095;

        proxy_set_header Host $host;

        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto https;
    }
